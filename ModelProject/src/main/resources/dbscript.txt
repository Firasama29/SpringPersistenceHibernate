create database websystique;

use websystique;

CREATE TABLE `user` (
   `ID` int(11) NOT NULL AUTO_INCREMENT,
   `NAME` varchar(50) DEFAULT NULL,
   `EMAIL` varchar(50) DEFAULT NULL,
   PRIMARY KEY (`ID`)
 ) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8

CREATE TABLE `USER_ADDRESS` (
   `USER_ADDRESS_ID` int(10) NOT NULL AUTO_INCREMENT,
   `USER_ID` int(10) NOT NULL,
   `ADDRESS1` varchar(255) CHARACTER SET latin1 DEFAULT NULL,
   `ADDRESS2` varchar(255) CHARACTER SET latin1 DEFAULT NULL,
   `ADDRESS3` varchar(255) CHARACTER SET latin1 DEFAULT NULL,
   `ADDRESS4` varchar(255) CHARACTER SET latin1 DEFAULT NULL,
   `ADDRESS5` varchar(255) CHARACTER SET latin1 DEFAULT NULL,
   `POSTCODE` varchar(10) CHARACTER SET latin1 DEFAULT NULL,
   `CITY` varchar(45) CHARACTER SET latin1 DEFAULT NULL,
   `STATE` varchar(45) CHARACTER SET latin1 DEFAULT NULL,
   `CNTRY` varchar(45) CHARACTER SET latin1 DEFAULT NULL,
   `TEL_NO` varchar(25) DEFAULT NULL,
   `CREATE_ID` varchar(20) CHARACTER SET latin1 DEFAULT NULL,
   `CREATE_DT` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
   `UPDATE_ID` varchar(20) CHARACTER SET latin1 DEFAULT NULL,
   `UPDATE_DT` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
   `STATUS` varchar(1) DEFAULT 'A',
   PRIMARY KEY (`USER_ADDRESS_IDuser_address`)
 ) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8
 
 
 
 

CREATE TABLE `CONTACT` (
   `CONTACT_ID` int(10) NOT NULL AUTO_INCREMENT,
   `USER_ID` int(10) NOT NULL,
   `POSTCODE` varchar(10) CHARACTER SET latin1 DEFAULT NULL,
   `CITY` varchar(45) CHARACTER SET latin1 DEFAULT NULL,
   `EMAIL` varchar(45) CHARACTER SET latin1 DEFAULT NULL,
   `TEL_NO` varchar(25) DEFAULT NULL,
   `CREATE_ID` varchar(20) CHARACTER SET latin1 DEFAULT NULL,
   PRIMARY KEY (`CONTACT_ID`)
 ) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8
